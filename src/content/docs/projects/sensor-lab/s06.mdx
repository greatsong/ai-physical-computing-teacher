```python
# ============================================
# 과학 탐구 6: 층별 기압 차이
# LPS22HB 센서로 1층~4층 기압을 측정하여
# 기압과 고도의 관계를 실측으로 확인
# ============================================

import network
import socket
import time
import json
import gc
from machine import Pin, I2C
from wifi_config import WIFI_SSID, WIFI_PASSWORD

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
while not wlan.isconnected():
    time.sleep(1)
my_ip = wlan.ifconfig()[0]
print(f"IP: &#123;my_ip&#125;")

# ── LPS22HB 드라이버 (I2C) ──────────────────
class LPS22HB:
    """LPS22HB 기압/온도 센서 (I2C 주소: 0x5C)"""
    # 레지스터 주소
    WHO_AM_I = 0x0F
    CTRL_REG1 = 0x10
    CTRL_REG2 = 0x11
    STATUS = 0x27
    PRESS_OUT_XL = 0x28
    PRESS_OUT_L = 0x29
    PRESS_OUT_H = 0x2A
    TEMP_OUT_L = 0x2B
    TEMP_OUT_H = 0x2C

    def __init__(self, i2c, addr=0x5C):
        self.i2c = i2c
        self.addr = addr

        # WHO_AM_I 확인 (0xB1이면 정상)
        who = self._read_reg(self.WHO_AM_I)
        if who != 0xB1:
            print(f"경고: WHO_AM_I = 0x&#123;who:02X&#125; (0xB1 예상)")
        else:
            print("✅ LPS22HB 확인됨")

        # 설정: 25Hz 연속 측정, BDU 활성화
        self._write_reg(self.CTRL_REG1, 0x30)
        time.sleep_ms(100)

    def _read_reg(self, reg):
        """레지스터 1바이트 읽기"""
        return self.i2c.readfrom_mem(self.addr, reg, 1)[0]

    def _write_reg(self, reg, val):
        """레지스터 1바이트 쓰기"""
        self.i2c.writeto_mem(self.addr, reg, bytes([val]))

    def read_pressure(self):
        """기압(hPa) 반환 — 24비트 읽기"""
        xl = self._read_reg(self.PRESS_OUT_XL)
        l = self._read_reg(self.PRESS_OUT_L)
        h = self._read_reg(self.PRESS_OUT_H)

        raw = (h << 16) | (l << 8) | xl
        # 음수 처리 (2의 보수)
        if raw & 0x800000:
            raw -= 0x1000000

        return raw / 4096.0  # hPa

    def read_temperature(self):
        """온도(°C) 반환 — 16비트 읽기"""
        l = self._read_reg(self.TEMP_OUT_L)
        h = self._read_reg(self.TEMP_OUT_H)

        raw = (h << 8) | l
        # 음수 처리 (2의 보수)
        if raw & 0x8000:
            raw -= 0x10000

        return raw / 100.0  # °C

    def read(self):
        """기압(hPa), 온도(°C) 반환"""
        p = round(self.read_pressure(), 2)
        t = round(self.read_temperature(), 1)
        return p, t

    def read_avg(self, n=10):
        """n회 측정 평균 (정밀도 향상)"""
        p_sum = 0
        t_sum = 0
        for _ in range(n):
            p, t = self.read()
            p_sum += p
            t_sum += t
            time.sleep_ms(50)
        return round(p_sum / n, 2), round(t_sum / n, 1)

# ── 센서 초기화 ─────────────────────────────
i2c = I2C(0, sda=Pin(4), scl=Pin(5), freq=100000)

# I2C 스캔
devices = i2c.scan()
print(f"I2C 장치: &#123;devices&#125;")

# 주소 자동 감지 (0x5C 또는 0x5D)
sensor_addr = 0x5C
if 0x5C in devices:
    sensor_addr = 0x5C
elif 0x5D in devices:
    sensor_addr = 0x5D
else:
    print("❌ LPS22HB 미감지! 배선 확인")

sensor = LPS22HB(i2c, sensor_addr)
time.sleep_ms(200)

# 테스트 읽기

# ── 고도 계산 ──────────────────────────────
def calc_altitude(p, p0):
    """기압으로 고도 계산 (m)
    h = 44330 * (1 - (P/P0)^0.1903)
    """
    if p0 <= 0 or p <= 0:
        return 0.0
    ratio = p / p0
    alt = 44330.0 * (1.0 - pow(ratio, 0.1903))
    return round(alt, 1)
```

## 정리

<div class="key-terms">
  <dl>
    <dt>LPS22HB</dt>
    <dd>MEMS 기압/온도 센서 (I2C 주소: 0x5C, 정확도: ±0.025 hPa)</dd>
    <dt>기압 (hPa)</dt>
    <dd>대기의 압력. 높이 올라갈수록 낮아짐 (1 hPa ≈ 고도 8.4m)</dd>
    <dt>고도 계산 공식</dt>
    <dd>h = 44330 × (1 - (P/P₀)^0.1903) — 국제 표준 대기 기준</dd>
    <dt>캘리브레이션</dt>
    <dd>기준점(1층)을 설정하여 상대적인 고도 차이를 계산하는 방법</dd>
    <dt>10회 평균 측정</dt>
    <dd>여러 번 측정한 평균을 사용하여 순간적 노이즈를 줄이는 기법</dd>
  </dl>
</div>

## 교사용 참고사항

### 실험 성공을 위한 팁

<div class="patrol-point">
  <ol>
    <li><strong>이동 후 안정화:</strong> 각 층에 도착하면 10초 이상 가만히 있다가 측정 (이동 중 기류 영향)</li>
    <li><strong>보조배터리 연결:</strong> Micro USB로 Pico에 전원 공급, 브레드보드 고정 주의</li>
    <li><strong>Wi-Fi 범위:</strong> 학교 Wi-Fi가 전 층에 커버되면 이상적. 안 되면 핫스팟 또는 Shell 기록</li>
    <li><strong>날씨 영향:</strong> 기압은 날씨에 따라 변하므로, 측정은 가급적 30분 이내에 완료</li>
    <li><strong>옥상 측정:</strong> 학교 옥상 접근 가능하면 더 큰 차이를 관찰할 수 있음</li>
  </ol>
</div>

### LPS22HB 센서 특성

| 항목 | 값 |
|------|------|
| I2C 주소 | 0x5C (또는 0x5D) |
| WHO_AM_I | 0xB1 |
| 측정 범위 | 260~1260 hPa |
| 정확도 | ±0.025 hPa (~0.2m 분해능) |
| 온도 범위 | -40~85°C |
| 측정 주기 | 최대 75Hz (코드에서 10회 평균 사용) |

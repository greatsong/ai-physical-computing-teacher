# ============================================
# 과학 탐구 5: 교실 CO2와 환기
# SCD30 센서로 CO2를 연속 측정하여
# 밀폐 교실에서의 CO2 상승 패턴 분석
# ============================================

import network
import socket
import time
import json
import struct
import gc
from machine import Pin, I2C
from wifi_config import WIFI_SSID, WIFI_PASSWORD

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
while not wlan.isconnected():
    time.sleep(1)
my_ip = wlan.ifconfig()[0]
print(f"IP: &#123;my_ip&#125;")

# ── SCD30 드라이버 (I2C) ────────────────────
class SCD30:
    """SCD30 CO2/온습도 센서 (I2C 주소: 0x61)"""
    def __init__(self, i2c, addr=0x61):
        self.i2c = i2c
        self.addr = addr
        time.sleep_ms(100)
        # 연속 측정 시작 (기압 보정 0 = 비활성)
        self._send_command(0x0010, 0)
        time.sleep_ms(20)

    def _send_command(self, cmd, arg=None):
        """명령 전송 (CRC 포함)"""
        data = bytearray([cmd >> 8, cmd & 0xFF])
        if arg is not None:
            data += bytearray([arg >> 8, arg & 0xFF])
            data += bytearray([self._crc(data[-2:])])
        self.i2c.writeto(self.addr, data)

    def _crc(self, data):
        """CRC-8 계산 (Sensirion 프로토콜)"""
        crc = 0xFF
        for byte in data:
            crc ^= byte
            for _ in range(8):
                if crc & 0x80:
                    crc = (crc << 1) ^ 0x31
                else:
                    crc <<= 1
                crc &= 0xFF
        return crc

    def _bytes_to_float(self, b):
        """4바이트 → float (IEEE 754 빅엔디안)"""
        return struct.unpack('>f', bytes(b))[0]

    def data_ready(self):
        """측정 데이터 준비 여부 확인"""
        self._send_command(0x0202)
        time.sleep_ms(3)
        data = self.i2c.readfrom(self.addr, 3)
        return data[1] == 1

    def read(self):
        """CO2(ppm), 온도(°C), 습도(%) 반환"""
        if not self.data_ready():
            return None, None, None

        self._send_command(0x0300)
        time.sleep_ms(3)
        data = self.i2c.readfrom(self.addr, 18)

        # 6바이트씩: [MSB, LSB, CRC, MSB, LSB, CRC]
        co2_bytes = [data[0], data[1], data[3], data[4]]
        temp_bytes = [data[6], data[7], data[9], data[10]]
        hum_bytes = [data[12], data[13], data[15], data[16]]

        co2 = round(self._bytes_to_float(co2_bytes), 0)
        temp = round(self._bytes_to_float(temp_bytes), 1)
        hum = round(self._bytes_to_float(hum_bytes), 1)

        return int(co2), temp, hum

# ── 센서 초기화 ─────────────────────────────
# SCD30은 50kHz에서 안정적 (100kHz 불안정)
i2c = I2C(0, sda=Pin(4), scl=Pin(5), freq=50000)

# I2C 스캔
devices = i2c.scan()
print(f"I2C 장치: &#123;devices&#125;")
if 0x61 not in devices:
    print("❌ SCD30 미감지! 배선 확인")
else:
    print("✅ SCD30 연결됨 (주소: 0x61)")

sensor = SCD30(i2c)
time.sleep(3)  # 첫 측정 대기 (2~5초 워밍업)

# 테스트 읽기
for _ in range(5):
    co2, temp, hum = sensor.read()
    if co2 is not None:
        print(f"현재: CO2=&#123;co2&#125;ppm, &#123;temp&#125;°C, &#123;hum&#125;%")
        break
    time.sleep(2)

# ── 데이터 저장소 ───────────────────────────
MAX_DATA = 120  # 최대 120개 (60분 @ 30초 간격)
data_log = []   # [&#123;"t": 초, "co2": ppm, "temp": °C, "hum": %&#125;, ...]
recording = False
start_time = 0

def add_data():
    """현재 센서값을 기록"""
    try:
        co2, temp, hum = sensor.read()
        if co2 is None:
            return None
        elapsed = time.ticks_diff(time.ticks_ms(), start_time) // 1000
        data_log.append(&#123;
            "t": elapsed,
            "co2": co2,
            "temp": temp,
            "hum": hum,
        &#125;)
        if len(data_log) > MAX_DATA:
            data_log.pop(0)
        return co2
    except Exception as e:
        print(f"센서 에러: &#123;e&#125;")
        return None

# ── HTML 페이지 ─────────────────────────────
def make_page():
    return """<!DOCTYPE html>
<html>
<head>

## 정리

<div class="key-terms">
  <dl>
    <dt>CO&#8322; (이산화탄소)</dt>
    <dd>사람이 호흡할 때 내뱉는 기체. 농도가 높으면 졸음, 두통 유발</dd>
    <dt>ppm (parts per million)</dt>
    <dd>백만 분의 1. 공기 중 CO&#8322; 비율을 나타내는 단위</dd>
    <dt>SCD30</dt>
    <dd>CO&#8322;, 온도, 습도를 동시에 측정하는 I2C 센서 (주소: 0x61)</dd>
    <dt>I2C 통신</dt>
    <dd>SDA(데이터)와 SCL(클록) 2개 선으로 여러 센서를 연결하는 방식</dd>
    <dt>환기 기준</dt>
    <dd>실내 CO&#8322; 1000ppm 이상이면 환기 권장 (한국 실내공기질 관리법)</dd>
  </dl>
</div>

## 교사용 참고사항

### 실험 성공을 위한 팁

<div class="patrol-point">
  <ol>
    <li><strong>시작 CO&#8322; 낮추기:</strong> 수업 시작 전 5분 환기하면 ~500ppm에서 출발 가능</li>
    <li><strong>밀폐가 핵심:</strong> 창문과 문을 확실히 닫아야 상승 효과가 뚜렷함</li>
    <li><strong>재미있는 데모:</strong> 센서 위에 직접 입김을 불면 5000ppm 이상으로 치솟음!</li>
    <li><strong>2000ppm 넘으면:</strong> 학생 안전을 위해 실험을 중단하고 환기하세요</li>
  </ol>
</div>

### SCD30 센서 특성

| 항목 | 값 |
|------|------|
| I2C 주소 | 0x61 |
| I2C 주파수 | 50kHz (100kHz 불안정) |
| CO&#8322; 범위 | 400~10000 ppm |
| 측정 주기 | 약 2초 (코드에서 30초 간격 기록) |
| 워밍업 | 첫 측정까지 2~5초 |
| 주의 | 센서가 비싸므로 취급 주의 안내 필수 |

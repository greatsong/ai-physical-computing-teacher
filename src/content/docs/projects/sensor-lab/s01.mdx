```python
# === DHT20 연결 확인 ===
from machine import Pin, I2C
import time

i2c = I2C(0, sda=Pin(4), scl=Pin(5), freq=100000)

devices = i2c.scan()
print(f"발견된 I2C 장치: {devices}")

if 0x38 in devices:
    print("✅ DHT20 연결 성공! (주소: 0x38)")
else:
    print("❌ DHT20이 감지되지 않습니다.")
    print("   → 배선을 확인하세요 (SDA↔GP4, SCL↔GP5)")
```

```python
# ============================================
# 과학 탐구 1: 에어컨과 습도
# DHT20 센서로 온습도를 연속 측정하여
# 시간-습도 그래프를 웹에서 실시간 확인
# ============================================

import network
import socket
import time
import json
import gc
from machine import Pin, I2C
from wifi_config import WIFI_SSID, WIFI_PASSWORD

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
while not wlan.isconnected():
    time.sleep(1)
my_ip = wlan.ifconfig()[0]
print(f"IP: {my_ip}")

# ── DHT20 드라이버 (I2C) ────────────────────
class DHT20:
    """DHT20 온습도 센서 (I2C 주소: 0x38)"""
    def __init__(self, i2c, addr=0x38):
        self.i2c = i2c
        self.addr = addr
        time.sleep_ms(100)
        # 초기화 확인
        status = self.i2c.readfrom(self.addr, 1)[0]
        if (status & 0x18) != 0x18:
            self.i2c.writeto(self.addr, bytes([0x71]))
            time.sleep_ms(10)

    def read(self):
        """온도(°C)와 습도(%) 반환"""
        # 측정 명령
        self.i2c.writeto(self.addr, bytes([0xAC, 0x33, 0x00]))
        time.sleep_ms(80)

        # 데이터 읽기
        data = self.i2c.readfrom(self.addr, 7)

        # 습도 계산
        hum_raw = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4))
        humidity = round(hum_raw / 1048576 * 100, 1)

        # 온도 계산
        temp_raw = ((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]
        temperature = round(temp_raw / 1048576 * 200 - 50, 1)

        return temperature, humidity
```

## 정리

<div class="key-terms">
  <dl>
    <dt>DHT20</dt>
    <dd>I2C 통신 방식의 온습도 센서 (주소: 0x38)</dd>
    <dt>I2C</dt>
    <dd>SDA(데이터)와 SCL(클록) 2선으로 통신하는 방식</dd>
    <dt>습도(%)</dt>
    <dd>공기 중 수증기의 비율. 에어컨은 냉각 과정에서 수분을 제거</dd>
    <dt>연속 측정</dt>
    <dd>일정 간격(10초)으로 반복 측정하여 변화 추이를 관찰</dd>
    <dt>가설 검증</dt>
    <dd>예측(가설)을 세우고, 데이터로 확인하는 과학적 방법</dd>
  </dl>
</div>

## 교사용 참고사항

### 실험 성공을 위한 팁

<div class="patrol-point">
  <ol>
    <li><strong>에어컨 효과가 빠르게 나타나려면:</strong> 센서를 에어컨 바람이 직접 닿는 곳에 배치</li>
    <li><strong>효과가 약하면:</strong> 창문/문을 닫아 밀폐 효과 높이기</li>
    <li><strong>습도가 안 변하면:</strong> 제습 모드가 아닌 냉방 모드인지 확인. 냉방도 습도를 낮추지만 제습보다는 느림</li>
    <li><strong>계절 영향:</strong> 겨울에는 실내 습도가 이미 낮아서 효과가 크지 않을 수 있음. 여름이 이상적</li>
  </ol>
</div>

### DHT20 센서 특성

| 항목 | 값 |
|------|------|
| I2C 주소 | 0x38 |
| 측정 범위 | 온도 -40~80°C, 습도 0~100% |
| 정확도 | 온도 ±0.3°C, 습도 ±2% |
| 측정 주기 | 최소 2초 권장 (코드에서 10초 간격 사용) |
| 풀업 저항 | 내장 (별도 필요 없음) |

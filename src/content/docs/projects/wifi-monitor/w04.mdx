# ============================================
# 교실 Wi-Fi 모니터 - 4차시
# 교실 Wi-Fi 신호 지도 (격자 측정 + SVG 히트맵)
# ============================================
import network
import socket
import time
import gc
from wifi_config import WIFI_SSID, WIFI_PASSWORD

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
print("Wi-Fi 연결 중...")
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
wait = 0
while not wlan.isconnected():
    time.sleep(1)
    wait += 1
    if wait > 15:
        print("연결 실패!")
        raise SystemExit
my_ip = wlan.ifconfig()[0]
print(f"연결 완료! IP: {my_ip}")

# ── 격자 설정 ───────────────────────────────
ROWS = 4
COLS = 5
LABELS = "ABCD"
grid = {}  # {"A1": -62, "B3": -71, ...}

# ── RSSI → 색상 변환 ────────────────────────
def rssi_color(v):
    v = max(-95, min(-30, v))
    hue = max(0, min(120, ((v + 95) * 120) // 65))
    return f"hsl({hue},85%,60%)"

# ── 측정 페이지 ──────────────────────────────
def page_measure(msg=""):
    gc.collect()
    rssi = wlan.status('rssi')
    filled = len(grid)
    total = ROWS * COLS
    sty = "display:flex;flex-direction:column;align-items:center;justify-content:center;height:60px;border-radius:12px;text-decoration:none;font-size:0.9em;"
    cells = ""
    for r in range(ROWS):
        for c in range(COLS):
            pos = f"{LABELS[r]}{c+1}"
            if pos in grid:
                val = grid[pos]
                bg = rssi_color(val)
                cells += f'<a href="/s?p={pos}" style="{sty}background:{bg};color:#000;font-weight:bold;">{pos}<br>{val}</a>'
            else:
                cells += f'<a href="/s?p={pos}" style="{sty}background:#1a1a3e;color:#888;border:2px dashed #444;">{pos}</a>'
    alert = ""
    if msg:
        alert = f'<div style="background:#1a3a1a;color:#51cf66;padding:10px;border-radius:8px;margin-bottom:12px;text-align:center;">{msg}</div>'
    return f"""<!DOCTYPE html>
<html><head><meta charset="UTF-8">
<meta name="viewport" content="width=device-width,initial-scale=1.0">
<title>측정 도우미</title></head>
<body style="font-family:-apple-system,sans-serif;background:#0f0f23;color:#ccc;padding:16px;margin:0;">
<div style="text-align:center;margin-bottom:16px;">
<h2 style="color:#00d4ff;margin:0;">측정 도우미</h2>
<p style="color:#888;font-size:0.85em;">현재 RSSI: <strong style="color:#ffd43b;">{rssi} dBm</strong> | 측정: {filled}/{total}</p></div>
{alert}
<div style="display:grid;grid-template-columns:repeat({COLS},1fr);gap:8px;max-width:400px;margin:0 auto;">
{cells}</div>
<div style="text-align:center;margin-top:16px;">
<a href="/map" style="color:#00d4ff;margin-right:16px;">히트맵 보기</a>
<a href="/reset" style="color:#ff6b6b;">초기화</a>
</div></body></html>"""

# ── 히트맵 페이지 ────────────────────────────
def page_map():
    gc.collect()
    if not grid:
        return """<!DOCTYPE html><html><head><meta charset="UTF-8">
<meta name="viewport" content="width=device-width,initial-scale=1.0">
<title>히트맵</title></head><body style="font-family:-apple-system,sans-serif;background:#0f0f23;color:#ccc;padding:16px;text-align:center;">
<h2 style="color:#00d4ff;">히트맵</h2>
<p>측정 데이터가 없습니다.</p>
<a href="/measure" style="color:#00d4ff;">측정하러 가기</a>
</body></html>"""
    vals = list(grid.values())
    best = max(vals)
    worst = min(vals)
    avg = sum(vals) // len(vals)
    dead = sum(1 for v in vals if v < -75)
    bp = [k for k, v in grid.items() if v == best][0]
    wp = [k for k, v in grid.items() if v == worst][0]
    cw, ch, pad = 70, 60, 30
    sw = COLS * cw + pad
    sh = ROWS * ch + pad
    svg = f'<svg viewBox="0 0 {sw} {sh}" style="max-width:400px;width:100%;">'
    for c in range(COLS):
        x = pad + c * cw + cw // 2
        svg += f'<text x="{x}" y="18" text-anchor="middle" fill="#888" font-size="14">{c+1}</text>'
    for r in range(ROWS):
        y = pad + r * ch
        svg += f'<text x="14" y="{y + ch // 2 + 5}" text-anchor="middle" fill="#888" font-size="14">{LABELS[r]}</text>'
        for c in range(COLS):
            x = pad + c * cw
            pos = f"{LABELS[r]}{c+1}"
            if pos in grid:
                val = grid[pos]
                color = rssi_color(val)
                svg += f'<rect x="{x}" y="{y}" width="{cw-2}" height="{ch-2}" rx="6" fill="{color}" opacity="0.85"/>'
                svg += f'<text x="{x+cw//2}" y="{y+ch//2-6}" text-anchor="middle" fill="#000" font-size="11" font-weight="bold">{pos}</text>'
                svg += f'<text x="{x+cw//2}" y="{y+ch//2+10}" text-anchor="middle" fill="#000" font-size="13" font-weight="bold">{val}</text>'
                if val == best:
                    svg += f'<rect x="{x}" y="{y}" width="{cw-2}" height="{ch-2}" rx="6" fill="none" stroke="#fff" stroke-width="2"/>'
            else:
                svg += f'<rect x="{x}" y="{y}" width="{cw-2}" height="{ch-2}" rx="6" fill="#1a1a3e" stroke="#333" stroke-dasharray="4"/>'
                svg += f'<text x="{x+cw//2}" y="{y+ch//2+4}" text-anchor="middle" fill="#555" font-size="11">{pos}</text>'
    svg += '</svg>'
    return f"""<!DOCTYPE html>
<html><head><meta charset="UTF-8">
<meta name="viewport" content="width=device-width,initial-scale=1.0">
<title>Wi-Fi 신호 지도</title></head>

    <li>측정 시간이 부족하면: 3x3 격자로 축소 또는 교사가 일부 데이터 사전 입력</li>

### 데이터 품질

| 상황 | 대응 |
|------|------|
| 같은 위치에서 ±5dBm 변동 | 정상 — 여러 번 측정 후 평균 사용 가능 |
| 극단적 이상값 (-10 또는 -99) | 재측정 안내 |
| 빈 칸이 있는 히트맵 | 시간 내에 채우기 어려우면 빈 칸 표시도 OK |
| 모둠 간 데이터 통합 | 각 모둠이 같은 Pico에 저장하거나, 수동으로 합치기 |

def get_data():
    """Pico에서 수집 가능한 모든 데이터를 딕셔너리로 반환"""
    gc.collect()

    rssi = wlan.status('rssi')
    uptime_sec = time.ticks_diff(time.ticks_ms(), start_time) // 1000
    mem_free = gc.mem_free()
    mem_alloc = gc.mem_alloc()
    mem_total = mem_free + mem_alloc
    mem_percent = round(mem_alloc / mem_total * 100, 1)

    # RSSI를 퍼센트(0~100%)로 변환
    # -30dBm = 100%, -90dBm = 0%
    rssi_percent = max(0, min(100, (rssi + 90) * 100 // 60))

    # 신호 품질 등급
    if rssi > -50:
        quality = "최고"
        quality_color = "#51cf66"
    elif rssi > -60:
        quality = "좋음"
        quality_color = "#94d82d"
    elif rssi > -70:
        quality = "보통"
        quality_color = "#ffd43b"
    elif rssi > -80:
        quality = "약함"
        quality_color = "#ff922b"
    else:
        quality = "매우 약함"
        quality_color = "#ff6b6b"

    return {
        "rssi": rssi,
        "rssi_percent": rssi_percent,
        "quality": quality,
        "quality_color": quality_color,
        "ip": my_ip,
        "uptime": uptime_sec,
        "mem_free": mem_free,
        "mem_percent": mem_percent,
        "visits": visit_count,
        "ssid": WIFI_SSID,
    }

def scan_ap():
    """주변 Wi-Fi AP를 스캔하여 리스트로 반환"""
    try:
        results = wlan.scan()
        ap_list = []
        for ap in results[:10]:  # 최대 10개
            ssid = ap[0].decode('utf-8', 'ignore')
            rssi = ap[3]
            channel = ap[2]
            # 보안 방식
            auth = ap[4]
            sec = ["OPEN", "WEP", "WPA-PSK", "WPA2-PSK", "WPA/WPA2"][min(auth, 4)]
            if ssid:  # 숨겨진 SSID 제외
                ap_list.append({
                    "ssid": ssid,
                    "rssi": rssi,
                    "ch": channel,
                    "sec": sec,
                })
        # RSSI 강한 순으로 정렬
        ap_list.sort(key=lambda x: x["rssi"], reverse=True)
        return ap_list
    except Exception as e:
        print(f"스캔 에러: {e}")
        return []

    try:

            # JSON 데이터 API
            visit_count += 1
            data = json.dumps(get_data())
            header = f"HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: {len(data)}\r\nConnection: close\r\n\r\n"

            client.send(data.encode())

        elif 'GET /scan' in request:
            # AP 스캔 API
            data = json.dumps(scan_ap())
            header = f"HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: {len(data)}\r\nConnection: close\r\n\r\n"

            client.send(data.encode())

        elif 'favicon' in request:
            client.close()
            continue

            # HTML 페이지
            page = make_page()
            header = f"HTTP/1.1 200 OK\r\nContent-Type: text/html; charset=utf-8\r\nContent-Length: {len(page.encode())}\r\nConnection: close\r\n\r\n"
            client.send(header.encode())
            client.send(page.encode())
            print(f"페이지 요청: {addr[0]}")

        client.close()

        print(f"에러: {e}")

            client.close()

### 메모리 관련 주의사항

<Aside type="caution" title="ENOMEM 에러 대응">
- HTML 코드가 길어지면 메모리 부족(`ENOMEM`) 발생 가능
- `gc.collect()`를 주기적으로 호출하여 메모리 회수
- 학생이 바이브 코딩으로 HTML을 많이 추가하면 메모리 이슈 가능
- 해결: HTML을 짧게 유지하거나, 큰 HTML은 여러 번 나눠 `send()`
</Aside>

### 흔한 문제 & 해결

| 증상 | 원인 | 해결 |
|------|------|------|
| 데이터 갱신 안 됨 | JavaScript 에러 | 브라우저 개발자 도구(F12) 확인, 코드 재실행 |
| AP 목록 "로딩 중..." | 스캔 실패 | Pico 재시작 후 재실행 |
| `ENOMEM` 에러 | HTML 코드 너무 큼 | HTML 간소화, gc.collect() 추가 |
| 접속 지연/타임아웃 | 스캔 중 요청 충돌 | 스캔 간격 늘리기 (30초) |
| JSON 파싱 에러 | 응답이 잘림 | Content-Length 확인, send 분할 |

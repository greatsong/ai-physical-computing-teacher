```python collapse={10-254}
# ============================================
# 교실 Wi-Fi 모니터 - 2차시
# 실시간 대시보드 + AP 스캔 (캐시)
# ============================================
import network
import socket
import time
import gc
from wifi_config import WIFI_SSID, WIFI_PASSWORD
# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
print("Wi-Fi 연결 중...")
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
wait = 0
while not wlan.isconnected():
    time.sleep(1)
    wait += 1
    if wait > 15:
        print("연결 실패!")
        raise SystemExit
my_ip = wlan.ifconfig()[0]
print(f"연결 완료! IP: {my_ip}")
visit_count = 0
start_time = time.ticks_ms()
# ── AP 스캔 (캐시) ──────────────────────────
ap_cache = '<tr><td colspan="4" style="color:#555;">스캔 중...</td></tr>'
last_scan = 0
SCAN_INTERVAL = 30
def scan_ap():
    try:
        results = wlan.scan()
        rows = []
        for ap in sorted(results, key=lambda x: x[3], reverse=True)[:8]:
            ssid = ap[0].decode('utf-8', 'ignore')
            if not ssid:
                continue
            rssi = ap[3]
            ch = ap[2]
            auth = ap[4]
            sec = ["OPEN", "WEP", "WPA", "WPA2", "WPA/2"][min(auth, 4)]
            pct = max(0, min(100, (rssi + 90) * 100 // 60))
            if rssi > -50:
                color = "#51cf66"
            elif rssi > -70:
                color = "#ffd43b"
            else:
                color = "#ff6b6b"
            mark = "⭐ " if ssid == WIFI_SSID else ""
            tc = "#00d4ff" if ssid == WIFI_SSID else "#ccc"
            rows.append(f'<tr><td style="color:{tc}">{mark}{ssid}</td>'
                        f'<td><span style="display:inline-block;height:6px;width:{pct}%;max-width:60px;background:{color};border-radius:3px;">&nbsp;</span> {rssi}</td>'
                        f'<td>{ch}</td><td>{sec}</td></tr>')
        return "\n".join(rows) if rows else '<tr><td colspan="4" style="color:#555;">결과 없음</td></tr>'
    except:
        return '<tr><td colspan="4" style="color:#555;">스캔 에러</td></tr>'
def get_ap_cached():
    global ap_cache, last_scan
    now = time.ticks_ms()
    if time.ticks_diff(now, last_scan) > SCAN_INTERVAL * 1000:
        ap_cache = scan_ap()
        last_scan = now
    return ap_cache

# ── 페이지 CSS (일반 문자열 — 중괄호 이스케이프 불필요) ──
PAGE_CSS = """<style>
    * { margin: 0; padding: 0; box-sizing: border-box; }
    body {
        font-family: -apple-system, 'Segoe UI', sans-serif;
        background: #0f0f23;
        color: #ccc;
        padding: 16px;
        min-height: 100vh;
    }
    .header { text-align: center; padding: 12px 0 16px; }
    .header h1 { color: #00d4ff; font-size: 1.4em; }
    .header .sub { color: #555; font-size: 0.8em; margin-top: 4px; }
    .dot {
        display: inline-block;
        width: 8px; height: 8px;
        background: #51cf66;
        border-radius: 50%;
        margin-right: 6px;
        animation: pulse 2s infinite;
    }
    @keyframes pulse {
        0%, 100% { opacity: 1; }
        50% { opacity: 0.4; }
    }
    .grid {
        display: grid;
        grid-template-columns: 1fr 1fr;
        gap: 10px;
        max-width: 500px;
        margin: 0 auto;
    }
    .card {
        background: #1a1a3e;
        border-radius: 12px;
        padding: 14px;
        text-align: center;
        border: 1px solid #2a2a5e;
    }
    .card.wide { grid-column: span 2; }
    .card .icon { font-size: 1.5em; margin-bottom: 6px; }
    .card .label { color: #888; font-size: 0.75em; margin-bottom: 2px; }
    .card .value { color: #00d4ff; font-size: 1.3em; font-weight: bold; }
    .card .unit { color: #666; font-size: 0.7em; }
    .bar-bg {
        width: 100%; height: 8px;
        background: #2a2a5e;
        border-radius: 4px;
        margin-top: 8px;
        overflow: hidden;
    }
    .bar { height: 100%; border-radius: 4px; }
    .ap-table {
        width: 100%;
        border-collapse: collapse;
        font-size: 0.8em;
        margin-top: 8px;
    }
    .ap-table th {
        color: #888; font-weight: normal;
        text-align: left; padding: 6px 4px;
        border-bottom: 1px solid #2a2a5e;
    }
    .ap-table td {
        padding: 5px 4px;
        border-bottom: 1px solid #1a1a3e;
    }
    .footer {
        text-align: center;
        margin-top: 16px;
        color: #333;
        font-size: 0.7em;
    }
</style>"""

# ── HTML 페이지 ─────────────────────────────
def make_page():
    gc.collect()
    rssi = wlan.status('rssi')
    uptime_sec = time.ticks_diff(time.ticks_ms(), start_time) // 1000
    mem_free = gc.mem_free()
    mem_alloc = gc.mem_alloc()
    mem_total = mem_free + mem_alloc
    mem_pct = round(mem_alloc / mem_total * 100, 1)
    rssi_pct = max(0, min(100, (rssi + 90) * 100 // 60))
    if rssi > -50:
        quality, qcolor = "최고", "#51cf66"
    elif rssi > -60:
        quality, qcolor = "좋음", "#94d82d"
    elif rssi > -70:
        quality, qcolor = "보통", "#ffd43b"
    elif rssi > -80:
        quality, qcolor = "약함", "#ff922b"
    else:
        quality, qcolor = "매우 약함", "#ff6b6b"
    if uptime_sec < 60:
        uptime = f"{uptime_sec}초"
    elif uptime_sec < 3600:
        uptime = f"{uptime_sec // 60}분 {uptime_sec % 60}초"
    else:
        uptime = f"{uptime_sec // 3600}시간 {(uptime_sec % 3600) // 60}분"
    ap_rows = get_ap_cached()
    return f"""<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="refresh" content="5">
    <title>Wi-Fi 모니터</title>
    {PAGE_CSS}
</head>
<body>
    <div class="header">
        <h1>Wi-Fi 모니터</h1>
        <div class="sub"><span class="dot"></span>실시간 모니터링 중 · 5초 자동 갱신</div>
    </div>
    <div class="grid">
        <div class="card wide">
            <div class="icon">📶</div>
            <div class="label">Wi-Fi 신호 세기</div>
            <div class="value">
                {rssi} <span class="unit">dBm</span>
                <span style="font-size:0.6em; margin-left:8px; color:{qcolor};">{quality}</span>
            </div>
            <div class="bar-bg">
                <div class="bar" style="width:{rssi_pct}%; background:{qcolor};"></div>
            </div>
        </div>
        <div class="card">
            <div class="icon">🌐</div>
            <div class="label">IP 주소</div>
            <div class="value" style="font-size:0.95em;">{my_ip}</div>
        </div>
        <div class="card">
            <div class="icon">⏱️</div>
            <div class="label">가동 시간</div>
            <div class="value">{uptime}</div>
        </div>
        <div class="card">
            <div class="icon">💾</div>
            <div class="label">메모리 사용</div>
            <div class="value">{mem_pct}%</div>
            <div class="unit">{mem_free:,} bytes free</div>
        </div>
        <div class="card">
            <div class="icon">👥</div>
            <div class="label">접속 횟수</div>
            <div class="value">{visit_count}</div>
        </div>
        <div class="card wide">
            <div class="icon">📋</div>
            <div class="label">주변 Wi-Fi 네트워크 (30초마다 갱신)</div>
            <table class="ap-table">
                <thead>
                    <tr><th>이름</th><th>신호</th><th>채널</th><th>보안</th></tr>
                </thead>
                <tbody>{ap_rows}</tbody>
            </table>
        </div>
    </div>
    <div class="footer">
        Powered by MicroPython on Pico 2 WH
    </div>
</body>
</html>"""
# ── 웹서버 시작 ─────────────────────────────
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
server.bind(('0.0.0.0', 80))
server.listen(2)
print()
print("=" * 45)
print(f"  대시보드 시작! http://{my_ip}")
print("=" * 45)
while True:
    try:
        client, addr = server.accept()
        request = client.recv(1024).decode()
        if 'favicon' in request:
            client.close()
            continue
        visit_count += 1
        print(f"접속 #{visit_count}: {addr[0]}")
        page = make_page()
        client.send(b"HTTP/1.1 200 OK\r\nContent-Type: text/html; charset=utf-8\r\nConnection: close\r\n\r\n")
        client.send(page.encode())
        client.close()
    except Exception as e:
        print(f"에러: {e}")
        try:
            client.close()
        except:
            pass
```

HTML `<meta http-equiv="refresh" content="5">`가 **5초마다** 페이지를 자동으로 다시 불러옵니다. Pico가 매번 최신 데이터를 HTML에 넣어서 보내주는 **서버사이드 렌더링** 방식이에요!
</Aside>

## 교사용 참고사항

### 서버사이드 렌더링 구조

이 코드는 JSON API가 아닌 **서버사이드 렌더링** 방식입니다. `make_page()` 함수가 모든 데이터를 수집하고, f-string으로 HTML에 직접 삽입하여 완성된 페이지를 전송합니다.

| 항목 | 함수/변수 | 설명 |
|------|-----------|------|
| RSSI | `wlan.status('rssi')` | make_page() 호출 시 측정 |
| AP 목록 | `get_ap_cached()` | 30초 캐시, HTML 테이블 행으로 반환 |
| 메모리 | `gc.mem_free()` | gc.collect() 후 측정 |
| 가동시간 | `time.ticks_diff()` | 시분초 형식 변환 |
| 자동갱신 | `<meta http-equiv="refresh">` | 5초 간격 |

### 메모리 관련 주의사항

<Aside type="caution" title="ENOMEM 에러 대응">
- HTML 코드가 길어지면 메모리 부족(`ENOMEM`) 발생 가능
- `gc.collect()`를 주기적으로 호출하여 메모리 회수
- 학생이 바이브 코딩으로 HTML을 많이 추가하면 메모리 이슈 가능
- 해결: HTML을 짧게 유지하거나, 큰 HTML은 여러 번 나눠 `send()`
</Aside>

### 흔한 문제 & 해결

| 증상 | 원인 | 해결 |
|------|------|------|
| 데이터 갱신 안 됨 | 브라우저 캐시 | 강제 새로고침 (Ctrl+Shift+R) |
| AP 목록 "스캔 중..." | 첫 30초 대기 중 | 30초 후 자동 갱신 |
| `ENOMEM` 에러 | HTML 코드 너무 큼 | HTML 간소화, CSS 축소 |
| 접속 지연/타임아웃 | Pico 부하 | Pico 재시작 후 재실행 |
| f-string 에러 | CSS 중괄호 이스케이프 누락 | CSS에서 `{{` `}}`로 이중 이스케이프 확인 |

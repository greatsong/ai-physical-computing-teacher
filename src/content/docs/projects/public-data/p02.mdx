# 공공데이터 전광판 - 2차시 기본 코드

import network
import socket
import time
import json
import gc
import ssl
from wifi_config import WIFI_SSID, WIFI_PASSWORD
from api_config import (
    NEIS_KEY, SCHOOL_CODE, ATPT_CODE, SCHOOL_NAME,
    WEATHER_KEY, WEATHER_NX, WEATHER_NY,
    AIR_KEY, AIR_STATION
)

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
print("Wi-Fi 연결 중...")
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
wait = 0
while not wlan.isconnected():
    time.sleep(1)
    wait += 1
    if wait > 15:
        raise SystemExit("연결 실패")
my_ip = wlan.ifconfig()[0]
print(f"연결 완료! IP: &#123;my_ip&#125;")

# ── NTP 시간 동기화 ─────────────────────────
try:
    import ntptime
    ntptime.settime()
    print("시간 동기화 완료")
except:
    print("시간 동기화 실패")

def kst():
    """현재 KST 시간 튜플"""
    return time.localtime(time.time() + 9 * 3600)

def today_str():
    t = kst()
    return f"&#123;t[0]&#125;&#123;t[1]:02d&#125;&#123;t[2]:02d&#125;"

def tomorrow_str():
    t = time.localtime(time.time() + 9 * 3600 + 86400)
    return f"&#123;t[0]&#125;&#123;t[1]:02d&#125;&#123;t[2]:02d&#125;"

# ── HTTPS 요청 함수 ─────────────────────────
def fetch_https(url):
    """HTTPS GET → 응답 본문 반환"""
    import socket as _s
    proto, _, hp = url.split("/", 2)
    host, path = hp.split("/", 1)
    path = "/" + path

    addr = _s.getaddrinfo(host, 443)[0][-1]
    s = _s.socket()
    s.settimeout(10)
    s.connect(addr)
    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
    ctx.verify_mode = ssl.CERT_NONE
    ss = ctx.wrap_socket(s, server_hostname=host)

    ss.write(f"GET &#123;path&#125; HTTP/1.0\r\nHost: &#123;host&#125;\r\nConnection: close\r\n\r\n".encode())
    data = b""
    while True:
        try:
            chunk = ss.read(1024)
            if not chunk:
                break
            data += chunk
        except:
            break
    ss.close()
    s.close()

    text = data.decode('utf-8', 'ignore')
    if "\r\n\r\n" in text:
        return text.split("\r\n\r\n", 1)[1]
    return text

def fetch_http(url):
    """HTTP GET (에어코리아용) → 응답 본문 반환"""
    import socket as _s
    _, _, hp = url.split("/", 2)
    if "/" in hp:
        host, path = hp.split("/", 1)
        path = "/" + path
    else:
        host = hp
        path = "/"

    addr = _s.getaddrinfo(host, 80)[0][-1]
    s = _s.socket()
    s.settimeout(10)
    s.connect(addr)
    s.send(f"GET &#123;path&#125; HTTP/1.0\r\nHost: &#123;host&#125;\r\nConnection: close\r\n\r\n".encode())

    data = b""
    while True:
        try:
            chunk = s.recv(1024)
            if not chunk:
                break
            data += chunk
        except:
            break
    s.close()

    text = data.decode('utf-8', 'ignore')
    if "\r\n\r\n" in text:
        return text.split("\r\n\r\n", 1)[1]
    return text

# ── 급식 API (1차시와 동일, 간소화) ─────────
def get_meal():
    date = today_str()
    url = (f"https://open.neis.go.kr/hub/mealServiceDietInfo"
           f"?KEY=&#123;NEIS_KEY&#125;&Type=json&pIndex=1&pSize=5"
           f"&ATPT_OFCDC_SC_CODE=&#123;ATPT_CODE&#125;"
           f"&SD_SCHUL_CODE=&#123;SCHOOL_CODE&#125;&MLSV_YMD=&#123;date&#125;")
    try:
        gc.collect()
        data = json.loads(fetch_https(url))
        meals = data.get("mealServiceDietInfo", [&#123;&#125;])
        if len(meals) < 2:
            return []
        rows = meals[1].get("row", [])
        for row in rows:
            if row.get("MMEAL_SC_CODE") == "2":
                raw = row.get("DDISH_NM", "")
                items = [x.strip() for x in raw.split("<br/>") if x.strip()]
                return items
        return []
    except Exception as e:
        print(f"급식 에러: &#123;e&#125;")
        return []

# ── 기상청 API ──────────────────────────────
def get_base_time_ultra():
    """초단기실황: 매시 정각 발표, 40분 이후 조회 가능"""
    t = kst()
    hour = t[3]
    minute = t[4]
    if minute < 40:
        hour -= 1
        if hour < 0:
            hour = 23
    return f"&#123;hour:02d&#125;00"

def get_base_time_fcst():
    """단기예보: 0200, 0500, 0800, 1100, 1400, 1700, 2000, 2300 발표"""
    t = kst()
    hour = t[3]
    bases = [23, 20, 17, 14, 11, 8, 5, 2]
    for b in bases:
        if hour >= b + 1:
            return f"&#123;b:02d&#125;00"
    return "2300"

def get_weather_now():
    """현재 기온, 습도, 강수 형태"""
    base_date = today_str()
    base_time = get_base_time_ultra()

    url = (f"https://apis.data.go.kr/1360000/VilageFcstInfoService_2.0/getUltraSrtNcst"
           f"?serviceKey=&#123;WEATHER_KEY&#125;&dataType=JSON&numOfRows=10"
           f"&base_date=&#123;base_date&#125;&base_time=&#123;base_time&#125;"
           f"&nx=&#123;WEATHER_NX&#125;&ny=&#123;WEATHER_NY&#125;")

    try:
        gc.collect()
        data = json.loads(fetch_https(url))
        items = data["response"]["body"]["items"]["item"]
        result = &#123;&#125;
        for item in items:
            cat = item["category"]
            val = item["obsrValue"]
            if cat == "T1H":
                result["temp"] = val
            elif cat == "REH":
                result["humidity"] = val
            elif cat == "PTY":
                result["pty"] = val
            elif cat == "RN1":
                result["rain"] = val
        return result
    except Exception as e:
        print(f"날씨 에러: &#123;e&#125;")
        return &#123;&#125;

def get_rain_forecast():
    """하교시간, 내일아침, 내일하교 강수확률"""
    base_date = today_str()
    base_time = get_base_time_fcst()
    tmr = tomorrow_str()

    url = (f"https://apis.data.go.kr/1360000/VilageFcstInfoService_2.0/getVilageFcst"
           f"?serviceKey=&#123;WEATHER_KEY&#125;&dataType=JSON&numOfRows=300"
           f"&base_date=&#123;base_date&#125;&base_time=&#123;base_time&#125;"
           f"&nx=&#123;WEATHER_NX&#125;&ny=&#123;WEATHER_NY&#125;")

    try:
        gc.collect()
        data = json.loads(fetch_https(url))
        items = data["response"]["body"]["items"]["item"]

        result = &#123;
            "today_16": "--",
            "tmr_08": "--",
            "tmr_16": "--",
        &#125;

        for item in items:
            if item["category"] != "POP":
                continue
            fd = item["fcstDate"]
            ft = item["fcstTime"]

            if fd == base_date and ft == "1600":
                result["today_16"] = item["fcstValue"]
            elif fd == tmr and ft == "0800":
                result["tmr_08"] = item["fcstValue"]
            elif fd == tmr and ft == "1600":
                result["tmr_16"] = item["fcstValue"]

        return result
    except Exception as e:
        print(f"예보 에러: &#123;e&#125;")
        return &#123;"today_16": "--", "tmr_08": "--", "tmr_16": "--"&#125;

# ── 에어코리아 API ──────────────────────────
def get_air():
    """미세먼지, 초미세먼지"""
    url = (f"http://apis.data.go.kr/B552584/ArpltnInforInqireSvc"
           f"/getMsrstnAcctoRltmMesureDnsty"

    try:
        gc.collect()
        data = json.loads(fetch_http(url))
        items = data["response"]["body"]["items"]
        if items:
            item = items[0]
            pm10 = item.get("pm10Value", "--")
            pm25 = item.get("pm25Value", "--")

            try:
                v = int(pm25)

                    grade_emoji = "🙂"

                    grade = "나쁨"
                    grade_color = "#ff922b"
                    grade_emoji = "😷"
                else:
                    grade = "매우나쁨"
                    grade_color = "#ff6b6b"
                    grade_emoji = "🤢"
            except:
                grade = "--"
                grade_color = "#888"
                grade_emoji = "❓"

            return &#123;
                "pm10": pm10, "pm25": pm25,
                "grade": grade, "grade_color": grade_color,
                "grade_emoji": grade_emoji,
            &#125;
        return &#123;&#125;
    except Exception as e:
        print(f"미세먼지 에러: &#123;e&#125;")
        return &#123;&#125;

# ── 전체 데이터 로드 ────────────────────────
print("데이터 로딩 중...")
meal_data = get_meal()

## 교사용 참고사항

### 기상청 API 상세

<Aside type="caution" title="기상청 API는 base_time이 핵심입니다">
- 초단기실황은 매시 정각 발표, 약 40분 뒤 API에 반영됩니다
- 단기예보는 하루 8회(02,05,08,11,14,17,20,23시) 발표됩니다
- base_time을 잘못 넣으면 "NO_DATA" 에러가 납니다
</Aside>

| API | 엔드포인트 | 발표 주기 | 주요 카테고리 |
|-----|-----------|----------|-------------|
| 초단기실황 `getUltraSrtNcst` | `/VilageFcstInfoService_2.0/getUltraSrtNcst` | 매시 정각 (40분 후 API) | T1H(기온), RN1(강수량), REH(습도), PTY(강수형태) |
| 단기예보 `getVilageFcst` | `/VilageFcstInfoService_2.0/getVilageFcst` | 하루 8회 | POP(강수확률), SKY(하늘상태), TMP(기온) |

**SKY 코드:** 1=맑음, 3=구름많음, 4=흐림

**PTY 코드:** 0=없음, 1=비, 2=비/눈, 3=눈, 4=소나기

### 에어코리아 API 상세

<Aside type="caution" title="에어코리아는 HTTP만 지원합니다">
HTTPS로 호출하면 연결이 실패합니다. 반드시 `http://`로 시작하는 URL을 사용하세요. 코드에서 `fetch_http()` 함수를 별도로 만든 이유입니다.
</Aside>

| 항목 | 내용 |
|------|------|
| 엔드포인트 | `http://apis.data.go.kr/B552584/ArpltnInforInqireSvc/getMsrstnAcctoRltmMesureDnsty` |
| 프로토콜 | **HTTP** (HTTPS 미지원) |
| 필수 파라미터 | `serviceKey`, `stationName=관악구`, `dataTerm=DAILY`, `returnType=json` |
| 주요 응답 필드 | `pm10Value`(미세먼지), `pm25Value`(초미세먼지) |

### PM2.5 등급 기준

| 등급 | PM2.5 (μg/m³) | 색상 |
|------|---------------|------|
| 좋음 | 0~15 | 초록 |
| 보통 | 16~35 | 노랑 |
| 나쁨 | 36~75 | 주황 |
| 매우나쁨 | 76~ | 빨강 |

### 트러블슈팅

| 증상 | 원인 | 해결 |
|------|------|------|
| 기상청 "NO_DATA" | 발표 전 시간 조회 | base_time 조정, 이전 발표 시간 사용 |
| 기상청 "SERVICE_KEY_IS_NOT_REGISTERED" | 키 미승인 또는 오타 | 키 재확인, 승인 대기 |
| 에어코리아 빈 응답 | HTTPS로 호출 또는 측정소 점검 | URL이 http://인지, fetch_http() 사용하는지 확인 |
| ENOMEM | 4개 API 순차 호출 메모리 부족 | 각 호출 사이 gc.collect() |
| 강수확률 "--" | 예보 시간 범위 밖 | numOfRows 늘리기, base_time 확인 |
| 전체 로딩 30초+ | HTTPS 4회 호출 | 정상 범위, 캐싱으로 이후 빠름 |

```python
# ============================================
# 공공데이터 전광판 - 1차시 기본 코드
# 우리 학교 급식 전광판 (나이스 API)
# ============================================

import network
import socket
import time
import json
import gc
import ssl
from wifi_config import WIFI_SSID, WIFI_PASSWORD
from api_config import NEIS_KEY, SCHOOL_CODE, ATPT_CODE, SCHOOL_NAME

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)

print("Wi-Fi 연결 중...")
wlan.connect(WIFI_SSID, WIFI_PASSWORD)

wait = 0
while not wlan.isconnected():
    time.sleep(1)
    wait += 1

def fetch_url(url):
    """HTTPS GET 요청을 보내고 응답 텍스트를 반환"""
    import socket as _socket

    # URL 파싱
    proto, _, host_path = url.split("/", 2)
    host, path = host_path.split("/", 1)
    path = "/" + path
    port = 443

    # 소켓 + SSL 연결
    addr = _socket.getaddrinfo(host, port)[0][-1]
    s = _socket.socket()
    s.settimeout(10)
    s.connect(addr)

    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
    ctx.verify_mode = ssl.CERT_NONE
    ss = ctx.wrap_socket(s, server_hostname=host)

    # HTTP 요청
    req = f"GET {path} HTTP/1.0\r\nHost: {host}\r\nConnection: close\r\n\r\n"
    ss.write(req.encode())

    # 응답 읽기
    data = b""
    while True:
        try:
            chunk = ss.read(1024)
            if not chunk:
                break
            data += chunk
        except:
            break
    ss.close()
    s.close()

    # 헤더와 본문 분리
    text = data.decode('utf-8', 'ignore')
    if "\r\n\r\n" in text:
        return text.split("\r\n\r\n", 1)[1]
    return text

def get_meal(date=None):
    """나이스 API에서 급식 정보를 가져옵니다"""
    if date is None:
        date = get_today()

    url = (
        f"https://open.neis.go.kr/hub/mealServiceDietInfo"
        f"?KEY={NEIS_KEY}"
        f"&Type=json&pIndex=1&pSize=5"
        f"&ATPT_OFCDC_SC_CODE={ATPT_CODE}"
        f"&SD_SCHUL_CODE={SCHOOL_CODE}"
        f"&MLSV_YMD={date}"
    )

    try:
        gc.collect()
        text = fetch_url(url)
        data = json.loads(text)

        meals = data.get("mealServiceDietInfo", [{}])
        if len(meals) < 2:
            return {"date": date, "menu": "급식 정보 없음", "cal": "", "type": ""}

        rows = meals[1].get("row", [])

        # 중식(2) 우선, 없으면 첫 번째
        meal = None
        for row in rows:
            if row.get("MMEAL_SC_CODE") == "2":
                meal = row
                break
        if meal is None and rows:
            meal = rows[0]

        if meal is None:
            return {"date": date, "menu": "급식 정보 없음", "cal": "", "type": ""}

        # 메뉴 파싱: <br/> 제거, 알레르기 번호 제거
        raw_menu = meal.get("DDISH_NM", "")
        menu_items = []
        for item in raw_menu.split("<br/>"):
            item = item.strip()
            if item:
                clean = ""
                i = 0
                while i < len(item):
                    if item[i].isdigit() or item[i] == '.':
                        j = i
                        while j < len(item) and (item[j].isdigit() or item[j] == '.'):
                            j += 1
                        if j == len(item) or item[j] in (' ', '\n', '\r'):
                            i = j
                            continue
                    clean += item[i]
                    i += 1
                clean = clean.strip()
                if clean:
                    menu_items.append(clean)

        cal = meal.get("CAL_INFO", "")
        meal_type = meal.get("MMEAL_SC_NM", "")

        return {
            "date": date,
            "menu": menu_items,
            "cal": cal,
            "type": meal_type,
        }
    except Exception as e:
        print(f"급식 API 에러: {e}")
        return {"date": date, "menu": "데이터를 가져올 수 없습니다", "cal": "", "type": ""}

def get_month_meals():
    """이번 달 전체 급식을 가져옵니다"""
    first, last = get_month_range()

    url = (
        f"https://open.neis.go.kr/hub/mealServiceDietInfo"
        f"?KEY={NEIS_KEY}"
        f"&Type=json&pIndex=1&pSize=100"
        f"&ATPT_OFCDC_SC_CODE={ATPT_CODE}"
        f"&SD_SCHUL_CODE={SCHOOL_CODE}"
        f"&MLSV_FROM_YMD={first}"
        f"&MLSV_TO_YMD={last}"
    )

    try:
        gc.collect()
        text = fetch_url(url)
        data = json.loads(text)

        meals = data.get("mealServiceDietInfo", [{}])
        if len(meals) < 2:
            return {}

        rows = meals[1].get("row", [])

        result = {}
        for row in rows:
            if row.get("MMEAL_SC_CODE") == "2":
                day = row.get("MLSV_YMD", "")
                raw = row.get("DDISH_NM", "")
                menu = raw.replace("<br/>", "\n").strip()
                result[day] = menu

        return result
    except Exception as e:
        print(f"월간 급식 에러: {e}")
        return {}
```

## 교사용 참고사항

### HTTPS 호출 관련

<Aside type="caution">
- Pico 2 WH의 최신 MicroPython(v1.24+)에서 `ssl` 모듈로 HTTPS 직접 호출 가능
- `ssl.CERT_NONE`으로 인증서 검증 생략 (교육 환경 한정)
- 첫 HTTPS 호출에 5~10초 소요 (TLS 핸드셰이크)
- 연속 호출 시 메모리 부족 가능 → `gc.collect()` 필수
- HTTPS가 불안정하면 교사 PC에 Flask 프록시를 띄우는 방법도 백업으로 준비
</Aside>

### 나이스 API 참고

| 항목 | 내용 |
|------|------|
| 엔드포인트 | `https://open.neis.go.kr/hub/mealServiceDietInfo` |
| 필수 파라미터 | `KEY`, `Type=json`, `ATPT_OFCDC_SC_CODE=B10`, `SD_SCHUL_CODE=7011536`, `MLSV_YMD=날짜` |
| 응답에서 급식 배열 | `mealServiceDietInfo[1].row[]` |
| 식사 코드 | `MMEAL_SC_CODE` — "1"(조식), "2"(중식), "3"(석식) |
| 메뉴 필드 | `DDISH_NM` — 알레르기번호 포함, `<br/>` 구분 |

### 흔한 문제 & 해결

| 증상 | 원인 | 해결 |
|------|------|------|
| `ssl` import 에러 | 펌웨어 버전 낮음 | MicroPython 1.24+ 업데이트 |
| API 응답 없음 / 타임아웃 | 네트워크 문제 | Wi-Fi 재연결, Pico 재시작 |
| JSON 파싱 에러 | 응답이 HTML(에러페이지) | API 키 확인, URL 파라미터 확인 |
| "급식 정보 없음" | 해당 날짜 급식 없음 | 주말/공휴일 확인, 날짜 변경 테스트 |
| ENOMEM | HTML + API 데이터 메모리 초과 | gc.collect() 추가, HTML 간소화 |
| 시간이 맞지 않음 | NTP 동기화 실패 | 수동으로 날짜 지정하여 테스트 |

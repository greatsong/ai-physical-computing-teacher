# 공공데이터 전광판 - 3차시 기본 코드

import network
import socket
import time
import json
import gc
import ssl
from wifi_config import WIFI_SSID, WIFI_PASSWORD
from api_config import (
    NEIS_KEY, SCHOOL_CODE, ATPT_CODE, SCHOOL_NAME,
    WEATHER_KEY, WEATHER_NX, WEATHER_NY,
    AIR_KEY, AIR_STATION,
    BUS_KEY, BUS_STATION_ID, BUS_STATION_NAME, BUS_ROUTE_NAME
)

# ── Wi-Fi 연결 ──────────────────────────────
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
print("Wi-Fi 연결 중...")
wlan.connect(WIFI_SSID, WIFI_PASSWORD)
wait = 0
while not wlan.isconnected():
    time.sleep(1)
    wait += 1
    if wait > 15:
        raise SystemExit("연결 실패")
my_ip = wlan.ifconfig()[0]
print(f"연결 완료! IP: {my_ip}")

# ── NTP ─────────────────────────────────────
try:
    import ntptime
    ntptime.settime()
except:
    pass

def kst():
    return time.localtime(time.time() + 9 * 3600)

def today_str():
    t = kst()
    return f"{t[0]}{t[1]:02d}{t[2]:02d}"

def tomorrow_str():
    t = time.localtime(time.time() + 9 * 3600 + 86400)
    return f"{t[0]}{t[1]:02d}{t[2]:02d}"

# ── HTTP/HTTPS 요청 ─────────────────────────
def fetch_https(url):
    import socket as _s
    _, _, hp = url.split("/", 2)
    host, path = hp.split("/", 1)
    path = "/" + path
    addr = _s.getaddrinfo(host, 443)[0][-1]
    s = _s.socket()
    s.settimeout(10)
    s.connect(addr)
    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
    ctx.verify_mode = ssl.CERT_NONE
    ss = ctx.wrap_socket(s, server_hostname=host)
    ss.write(f"GET {path} HTTP/1.0\r\nHost: {host}\r\nConnection: close\r\n\r\n".encode())
    data = b""
    while True:
        try:
            chunk = ss.read(1024)
            if not chunk: break
            data += chunk
        except: break
    ss.close(); s.close()
    text = data.decode('utf-8', 'ignore')
    return text.split("\r\n\r\n", 1)[1] if "\r\n\r\n" in text else text

def fetch_http(url):
    import socket as _s
    _, _, hp = url.split("/", 2)
    if "/" in hp:
        host, path = hp.split("/", 1)
        path = "/" + path
    else:
        host, path = hp, "/"
    addr = _s.getaddrinfo(host, 80)[0][-1]
    s = _s.socket()
    s.settimeout(10)
    s.connect(addr)
    s.send(f"GET {path} HTTP/1.0\r\nHost: {host}\r\nConnection: close\r\n\r\n".encode())
    data = b""
    while True:
        try:
            chunk = s.recv(1024)
            if not chunk: break
            data += chunk
        except: break
    s.close()
    text = data.decode('utf-8', 'ignore')
    return text.split("\r\n\r\n", 1)[1] if "\r\n\r\n" in text else text

# ── 급식 (간소화) ───────────────────────────
def get_meal():
    url = (f"https://open.neis.go.kr/hub/mealServiceDietInfo"
           f"?KEY={NEIS_KEY}&Type=json&pIndex=1&pSize=5"
           f"&ATPT_OFCDC_SC_CODE={ATPT_CODE}"
           f"&SD_SCHUL_CODE={SCHOOL_CODE}&MLSV_YMD={today_str()}")
    try:
        gc.collect()
        data = json.loads(fetch_https(url))
        meals = data.get("mealServiceDietInfo", [{}])
        if len(meals) < 2: return []
        for row in meals[1].get("row", []):
            if row.get("MMEAL_SC_CODE") == "2":
                return [x.strip() for x in row.get("DDISH_NM", "").split("<br/>") if x.strip()]
        return []
    except Exception as e:
        print(f"급식 에러: {e}"); return []

# ── 날씨 ────────────────────────────────────
def get_base_time_ultra():
    t = kst()
    h = t[3] - (1 if t[4] < 40 else 0)
    if h < 0: h = 23
    return f"{h:02d}00"

def get_base_time_fcst():
    h = kst()[3]
    for b in [23,20,17,14,11,8,5,2]:
        if h >= b + 1: return f"{b:02d}00"
    return "2300"

def get_weather():
    url = (f"https://apis.data.go.kr/1360000/VilageFcstInfoService_2.0/getUltraSrtNcst"
           f"?serviceKey={WEATHER_KEY}&dataType=JSON&numOfRows=10"
           f"&base_date={today_str()}&base_time={get_base_time_ultra()}"
           f"&nx={WEATHER_NX}&ny={WEATHER_NY}")
    try:
        gc.collect()
        items = json.loads(fetch_https(url))["response"]["body"]["items"]["item"]
        r = {}
        for i in items:
            c, v = i["category"], i["obsrValue"]
            if c == "T1H": r["temp"] = v
            elif c == "REH": r["hum"] = v
            elif c == "PTY": r["pty"] = v
        return r
    except Exception as e:
        print(f"날씨 에러: {e}"); return {}

def get_rain():
    url = (f"https://apis.data.go.kr/1360000/VilageFcstInfoService_2.0/getVilageFcst"
           f"?serviceKey={WEATHER_KEY}&dataType=JSON&numOfRows=300"
           f"&base_date={today_str()}&base_time={get_base_time_fcst()}"
           f"&nx={WEATHER_NX}&ny={WEATHER_NY}")
    try:
        gc.collect()
        items = json.loads(fetch_https(url))["response"]["body"]["items"]["item"]
        r = {"t16": "--", "t08": "--", "t16t": "--"}
        tmr = tomorrow_str()
        for i in items:
            if i["category"] != "POP": continue
            fd, ft = i["fcstDate"], i["fcstTime"]
            if fd == today_str() and ft == "1600": r["t16"] = i["fcstValue"]
            elif fd == tmr and ft == "0800": r["t08"] = i["fcstValue"]
            elif fd == tmr and ft == "1600": r["t16t"] = i["fcstValue"]
        return r
    except Exception as e:
        print(f"예보 에러: {e}"); return {"t16":"--","t08":"--","t16t":"--"}

# ── 미세먼지 ────────────────────────────────
def get_air():
    url = (f"http://apis.data.go.kr/B552584/ArpltnInforInqireSvc"
           f"/getMsrstnAcctoRltmMesureDnsty"
           f"?serviceKey={AIR_KEY}&returnType=json&numOfRows=1"
           f"&stationName={AIR_STATION}&dataTerm=DAILY&ver=1.0")
    try:
        gc.collect()
        item = json.loads(fetch_http(url))["response"]["body"]["items"][0]
        pm25 = item.get("pm25Value", "--")
        try:
            v = int(pm25)
            if v <= 15: return {"pm25": pm25, "g": "좋음", "gc": "#51cf66", "ge": "😊"}
            elif v <= 35: return {"pm25": pm25, "g": "보통", "gc": "#94d82d", "ge": "🙂"}
            elif v <= 75: return {"pm25": pm25, "g": "나쁨", "gc": "#ff922b", "ge": "😷"}
            else: return {"pm25": pm25, "g": "매우나쁨", "gc": "#ff6b6b", "ge": "🤢"}
        except:
            return {"pm25": pm25, "g": "--", "gc": "#888", "ge": "❓"}
    except Exception as e:
        print(f"대기질 에러: {e}"); return {"pm25":"--","g":"--","gc":"#888","ge":"❓"}

# ── 버스 도착정보 (NEW!) ────────────────────
def get_bus():
    """관악11번 버스 도착 정보"""
    url = (f"http://ws.bus.go.kr/api/rest/stationinfo/getStationByUid"
           f"?ServiceKey={BUS_KEY}&arsId={BUS_STATION_ID}")
    try:
        gc.collect()
        xml = fetch_http(url)

        # 간단한 XML 파싱 (관악11 노선만 추출)
        buses = []

        parts = xml.split("<itemList>")
        for part in parts[1:]:
            block = part.split("</itemList>")[0]

            rt = ""
            if "<rtNm>" in block:
                rt = block.split("<rtNm>")[1].split("</rtNm>")[0]

            if BUS_ROUTE_NAME not in rt:
                continue

            msg1 = msg2 = ""
            if "<arrmsg1>" in block:
                msg1 = block.split("<arrmsg1>")[1].split("</arrmsg1>")[0]
            if "<arrmsg2>" in block:
                msg2 = block.split("<arrmsg2>")[1].split("</arrmsg2>")[0]

            t1 = t2 = 0
            if "<traTime1>" in block:
                try: t1 = int(block.split("<traTime1>")[1].split("</traTime1>")[0])
                except: pass
            if "<traTime2>" in block:
                try: t2 = int(block.split("<traTime2>")[1].split("</traTime2>")[0])
                except: pass

            buses.append({
                "route": rt,
                "msg1": msg1,
                "msg2": msg2,
                "sec1": t1,
                "sec2": t2,
            })

        if buses:
            return buses[0]
        return {"route": BUS_ROUTE_NAME, "msg1": "도착 정보 없음", "msg2": "", "sec1": 0, "sec2": 0}

    except Exception as e:
        print(f"버스 에러: {e}")
        return {"route": BUS_ROUTE_NAME, "msg1": "정보 조회 실패", "msg2": "", "sec1": 0, "sec2": 0}

# ── 전체 데이터 로드 ────────────────────────
print("=== 데이터 로딩 ===")
meal_data = get_meal(); print(f"  급식: {len(meal_data)}개"); gc.collect()
weather_data = get_weather(); print(f"  날씨: {weather_data.get('temp','?')}°C"); gc.collect()

# ── HTML ────────────────────────────────────
def make_page():

    date_str = f"{t[1]}월 {t[2]}일"
    wd = ["월","화","수","목","금","토","일"][t[6]]

    # 강수형태

    pty = pty_map.get(weather_data.get("pty","0"), "")

    # 기온 이모지
    try:
        tv = float(weather_data.get("temp", 20))

## 교사용 참고사항

### 서울시 버스 API 상세

| 항목 | 내용 |
|------|------|
| 엔드포인트 | `http://ws.bus.go.kr/api/rest/stationinfo/getStationByUid` |
| 파라미터 | `ServiceKey`, `arsId` |
| 응답 형식 | XML |
| 주요 태그 | `<rtNm>` 노선명, `<arrmsg1>` 도착 메시지, `<traTime1>` 도착 시간(초) |
| 프로토콜 | HTTP (HTTPS 아님) |
| 참고 | 정류장에 여러 노선이 정차 → rtNm으로 "관악11" 필터링 필요 |

### 흔한 문제 & 해결

| 증상 | 원인 | 해결 |
|------|------|------|
| 버스 "도착 정보 없음" | 운행 시간 외 | 운행 시간 확인 (보통 06:00~23:00) |
| 버스 API XML 파싱 실패 | 응답 형식 변경 | AI에게 XML 파싱 개선 요청 |
| 전체 로딩 40초+ | API 5개 순차 호출 | 정상 범위, 캐싱으로 이후 빠름 |
| ENOMEM 빈발 | HTML 코드 길이 증가 | CSS 변수명 축약, 불필요한 요소 제거 |
| 버스 데이터 갱신 느림 | 30초 주기 | 정상, 더 빠르게 하면 API 제한 |

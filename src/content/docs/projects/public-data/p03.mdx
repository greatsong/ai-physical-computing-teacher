def fetch_https(url):
    import socket as _s
    _, _, hp = url.split("/", 2)
    host, path = hp.split("/", 1)
    path = "/" + path
    addr = _s.getaddrinfo(host, 443)[0][-1]
    s = _s.socket()
    s.settimeout(10)
    s.connect(addr)
    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
    ctx.verify_mode = ssl.CERT_NONE
    ss = ctx.wrap_socket(s, server_hostname=host)
    ss.write(f"GET &#123;path&#125; HTTP/1.0\r\nHost: &#123;host&#125;\r\nConnection: close\r\n\r\n".encode())
    data = b""
    while True:
        try:
            chunk = ss.read(1024)
            if not chunk: break
            data += chunk
        except: break
    ss.close(); s.close()
    text = data.decode('utf-8', 'ignore')
    return text.split("\r\n\r\n", 1)[1] if "\r\n\r\n" in text else text

def fetch_http(url):
    import socket as _s
    _, _, hp = url.split("/", 2)
    if "/" in hp:
        host, path = hp.split("/", 1)
        path = "/" + path
    else:
        host, path = hp, "/"
    addr = _s.getaddrinfo(host, 80)[0][-1]
    s = _s.socket()
    s.settimeout(10)
    s.connect(addr)
    s.send(f"GET &#123;path&#125; HTTP/1.0\r\nHost: &#123;host&#125;\r\nConnection: close\r\n\r\n".encode())
    data = b""
    while True:
        try:
            chunk = s.recv(1024)
            if not chunk: break
            data += chunk
        except: break
    s.close()
    text = data.decode('utf-8', 'ignore')
    return text.split("\r\n\r\n", 1)[1] if "\r\n\r\n" in text else text

# ── 버스 도착정보 (XML 파싱) ─────────────────
def get_bus():
    """관악11번 버스 도착 정보 (XML 응답)"""
    url = (f"http://ws.bus.go.kr/api/rest/stationinfo/getStationByUid"
           f"?ServiceKey=&#123;BUS_KEY&#125;&arsId=&#123;BUS_STATION_ID&#125;")
    try:
        gc.collect()
        xml = fetch_http(url)

        # 간단한 XML 파싱: itemList 블록 분리 → 관악11 필터링
        parts = xml.split("<itemList>")
        for part in parts[1:]:
            block = part.split("</itemList>")[0]

            # 노선명 추출
            rt = ""
            if "<rtNm>" in block:
                rt = block.split("<rtNm>")[1].split("</rtNm>")[0]

            if BUS_ROUTE_NAME not in rt:
                continue

            # 도착 메시지 추출
            msg1 = msg2 = ""
            if "<arrmsg1>" in block:
                msg1 = block.split("<arrmsg1>")[1].split("</arrmsg1>")[0]
            if "<arrmsg2>" in block:
                msg2 = block.split("<arrmsg2>")[1].split("</arrmsg2>")[0]

            # 도착 예정 시간(초) 추출
            t1 = t2 = 0
            if "<traTime1>" in block:
                try: t1 = int(block.split("<traTime1>")[1].split("</traTime1>")[0])
                except: pass
            if "<traTime2>" in block:
                try: t2 = int(block.split("<traTime2>")[1].split("</traTime2>")[0])
                except: pass

            return &#123;
                "route": rt, "msg1": msg1, "msg2": msg2,

## 교사용 참고사항

### 서울시 버스 도착 API 상세

<Aside type="caution" title="버스 API도 HTTP만 지원합니다">
에어코리아와 마찬가지로 HTTPS가 아닌 HTTP로 호출해야 합니다. 코드에서 `fetch_http()` 함수를 사용합니다. 응답 형식은 JSON이 아닌 XML입니다.
</Aside>

| 항목 | 내용 |
|------|------|
| 엔드포인트 | `http://ws.bus.go.kr/api/rest/stationinfo/getStationByUid` |
| 프로토콜 | **HTTP** (HTTPS 미지원) |
| 필수 파라미터 | `ServiceKey`, `arsId=20522` |
| 응답 형식 | **XML** (JSON 아님) |
| 주요 응답 태그 | `rtNm`(노선명), `arrmsg1`(첫번째 도착), `arrmsg2`(두번째 도착), `traTime1`/`traTime2`(도착예정 초) |

### XML 응답 구조 예시

```
<itemList>
  <rtNm>관악11</rtNm>
  <arrmsg1>3분12초후[1번째 전]</arrmsg1>
  <arrmsg2>8분5초후[3번째 전]</arrmsg2>
  <traTime1>192</traTime1>
  <traTime2>485</traTime2>
</itemList>
```

### 갱신 주기 전략

| 데이터 | 갱신 주기 | 이유 |
|--------|----------|------|
| 버스 도착 | 30초 | 실시간 데이터, 빠르게 변함 |
| 날씨·미세먼지 | 10분 | 발표 주기가 길고 급변하지 않음 |
| 급식 | 시작시 1회 | 하루 동안 변하지 않음 |

### 트러블슈팅

| 증상 | 원인 | 해결 |
|------|------|------|
| 버스 '도착 정보 없음' | 운행 시간 외 (심야/새벽) | 운행 시간에 테스트 (보통 06:00~23:00) |
| 버스 API 연결 실패 | HTTPS로 호출 | URL이 `http://ws.bus.go.kr`인지, fetch_http() 사용하는지 확인 |
| XML 파싱 에러 | 태그 이름 불일치 | Shell에서 원본 XML print해서 태그 확인 |
| 다른 노선만 나옴 | BUS_ROUTE_NAME 불일치 | api_config.py에서 정확한 노선명 확인 |
| ENOMEM 빈발 | API 5개 순차 호출 + HTML 길이 | 각 호출 사이 gc.collect(), CSS 변수명 축약 |
| 30초 갱신이 느림 | HTTP 요청 오버헤드 | 정상 범위. HTTP는 HTTPS보다 빠름 |

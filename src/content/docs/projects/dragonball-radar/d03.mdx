```python
# ============================================
# 🌐 피코 웹 서버 — 기초 버전
# ============================================
# 피코가 AP를 열고, 접속한 스마트폰에
# 간단한 웹 페이지를 보여줍니다.
# ============================================

import network
import socket
import time

# === AP 모드 설정 ===
RADAR_AP_NAME = "DragonRadar"
RADAR_AP_PASS = "radarpass"

ap = network.WLAN(network.AP_IF)
ap.active(True)
ap.config(essid=RADAR_AP_NAME, password=RADAR_AP_PASS)

while not ap.active():
    time.sleep(1)

print(f"📡 AP 활성화: &#123;RADAR_AP_NAME&#125;")
print(f"   IP: &#123;ap.ifconfig()[0]&#125;")

# === 웹 페이지 HTML ===
def make_html():
    return """<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <title>Dragon Radar</title>
    <style>
        body &#123;
            font-family: Arial, sans-serif;
            background: #1a1a2e;
            color: #eee;
            text-align: center;
            padding: 20px;
        &#125;
        h1 &#123; color: #ff6b35; font-size: 2em; &#125;
        .status &#123;
            font-size: 1.5em;
            margin: 30px;
            padding: 20px;
            background: #16213e;
            border-radius: 15px;
        &#125;
    </style>
</head>
<body>
    <h1>🐉 Dragon Radar</h1>
    <div class="status">
        ✅ 레이더 서버 작동 중!<br>
        접속 성공!
    </div>
    <p>다음 단계에서 실시간 데이터가 여기에 표시됩니다.</p>
</body>
</html>"""

# === 웹 서버 시작 ===
addr = socket.getaddrinfo('0.0.0.0', 80)[0][-1]
s = socket.socket()
s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
s.bind(addr)
s.listen(1)

print(f"🌐 웹 서버 시작! http://&#123;ap.ifconfig()[0]&#125;")
print("   스마트폰으로 위 주소에 접속하세요!")

while True:
    try:
        cl, addr = s.accept()
        print(f"   📱 접속: &#123;addr[0]&#125;")

        request = cl.recv(1024)

        response = make_html()
        cl.send('HTTP/1.0 200 OK\r\n')
        cl.send('Content-type: text/html\r\n')
        cl.send('\r\n')
        cl.send(response)
        cl.close()
    except Exception as e:
        print(f"   에러: &#123;e&#125;")
        try:
            cl.close()
        except:
            pass
```

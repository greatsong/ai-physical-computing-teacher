```python
# ============================================
# 드래곤볼 LED 레이더 - 기본 버전
# ============================================
# RSSI 값에 따라 온보드 LED 깜빡임 속도가 변합니다.
# 드래곤볼에 가까울수록 빠르게 깜빡!
# ============================================

import network
import machine
import time

# === 설정 ===
TARGET_DB = "DB1"          # 찾을 드래곤볼 이름
RSSI_MIN = -90             # 가장 약한 신호 (멀리)
RSSI_MAX = -30             # 가장 강한 신호 (가까이)
DELAY_SLOW = 1.0           # 멀 때 깜빡임 간격 (초)
DELAY_FAST = 0.05          # 가까울 때 깜빡임 간격 (초)

# === LED 준비 ===
led = machine.Pin("LED", machine.Pin.OUT)

# === STA 모드 켜기 ===
sta = network.WLAN(network.STA_IF)
sta.active(True)

# === 매핑 함수 ===
# 한 범위의 값을 다른 범위로 변환
def map_value(value, in_min, in_max, out_min, out_max):
    """
    value:   변환할 값 (예: RSSI = -55)
    in_min:  입력 범위 최소값 (예: -90)
    in_max:  입력 범위 최대값 (예: -30)
    out_min: 출력 범위 최소값 (예: 1.0)
    out_max: 출력 범위 최대값 (예: 0.05)
    """
    # 입력값을 범위 안으로 제한 (clamp)
    value = max(in_min, min(in_max, value))

    # 비례식으로 변환
    result = out_min + (value - in_min) * (out_max - out_min) / (in_max - in_min)
    return result

# === 드래곤볼 RSSI 찾기 함수 ===
def find_dragonball(target_name):
    """주변 스캔 -> 특정 드래곤볼의 RSSI 반환. 없으면 None."""
    networks = sta.scan()
    for net in networks:
        ssid = net[0].decode('utf-8')
        rssi = net[3]
        if ssid == target_name:
            return rssi
    return None

# === 메인 루프 ===
print(f"드래곤볼 [&#123;TARGET_DB&#125;] 레이더 가동!")
print(f"   신호 범위: &#123;RSSI_MIN&#125; ~ &#123;RSSI_MAX&#125; dBm")
print(f"   깜빡임: &#123;DELAY_SLOW&#125;초(멀리) ~ &#123;DELAY_FAST&#125;초(가까이)")
print("=" * 50)

while True:
    rssi = find_dragonball(TARGET_DB)

    if rssi is not None:
        # RSSI -> 딜레이 변환
        delay = map_value(rssi, RSSI_MIN, RSSI_MAX, DELAY_SLOW, DELAY_FAST)

        print(f"  드래곤볼 &#123;TARGET_DB&#125;: &#123;rssi&#125;dBm -> 딜레이 &#123;delay:.3f&#125;초")

        # LED 깜빡이기
        led.on()
        time.sleep(delay)
        led.off()
        time.sleep(delay)
    else:
        # 드래곤볼을 찾지 못함 -> LED 꺼짐
        led.off()
        print(f"  &#123;TARGET_DB&#125; 감지 안 됨...")
        time.sleep(1)
```

```python
# ============================================
# 드래곤볼 LED 레이더 - 이동 평균 버전
# ============================================
# 이동 평균(Moving Average)으로 RSSI 노이즈를 줄여
# LED 깜빡임이 훨씬 부드럽고 안정적입니다.
# ============================================

import network
import machine
import time

# === 설정 ===
TARGET_DB = "DB1"          # 찾을 드래곤볼
RSSI_MIN = -90             # 신호 범위 최소
RSSI_MAX = -30             # 신호 범위 최대
DELAY_SLOW = 1.0           # 멀 때 딜레이
DELAY_FAST = 0.05          # 가까울 때 딜레이

# NEW 이동 평균 설정
WINDOW_SIZE = 5            # 여기를 바꿔보세요! (3, 5, 7, 10)

# === LED & STA 준비 ===
led = machine.Pin("LED", machine.Pin.OUT)
sta = network.WLAN(network.STA_IF)
sta.active(True)

# === 매핑 함수 (1교시와 동일) ===
def map_value(value, in_min, in_max, out_min, out_max):
    value = max(in_min, min(in_max, value))
    return out_min + (value - in_min) * (out_max - out_min) / (in_max - in_min)

# === 드래곤볼 찾기 함수 (1교시와 동일) ===
def find_dragonball(target_name):
    networks = sta.scan()
    for net in networks:
        ssid = net[0].decode('utf-8')
        rssi = net[3]
        if ssid == target_name:
            return rssi
    return None

# === NEW 이동 평균 클래스 ===
class MovingAverage:
    """최근 N개 값의 평균을 계산하는 버퍼"""

    def __init__(self, window_size):
        self.window = window_size    # 창 크기
        self.buffer = []             # 값 저장 리스트

    def add(self, value):
        """새 값 추가. 창 크기 초과하면 오래된 값 제거."""
        self.buffer.append(value)
        if len(self.buffer) > self.window:
            self.buffer.pop(0)       # 가장 오래된 값 빼기

    def average(self):
        """현재 버퍼의 평균값 반환."""
        if not self.buffer:
            return None
        return sum(self.buffer) / len(self.buffer)

    def count(self):
        """현재 버퍼에 있는 값의 개수."""
        return len(self.buffer)

# === 이동 평균 객체 생성 ===
rssi_smoother = MovingAverage(WINDOW_SIZE)

# === 메인 루프 ===
print(f"드래곤볼 [&#123;TARGET_DB&#125;] 레이더 (이동평균 ver.)")
print(f"   이동 평균 창 크기: &#123;WINDOW_SIZE&#125;")
print("=" * 50)

while True:
    rssi_raw = find_dragonball(TARGET_DB)

    if rssi_raw is not None:
        # NEW 이동 평균 적용
        rssi_smoother.add(rssi_raw)
        rssi_smooth = rssi_smoother.average()

        # 부드러워진 RSSI -> 딜레이 변환
        delay = map_value(rssi_smooth, RSSI_MIN, RSSI_MAX, DELAY_SLOW, DELAY_FAST)

        print(f"  원본: &#123;rssi_raw&#125;dBm | "
              f"평균: &#123;rssi_smooth:.1f&#125;dBm "
              f"(&#123;rssi_smoother.count()&#125;/&#123;WINDOW_SIZE&#125;) | "
              f"딜레이: &#123;delay:.3f&#125;초")

        # LED 깜빡이기
        led.on()
        time.sleep(delay)
        led.off()
        time.sleep(delay)
    else:
        led.off()
        print(f"  &#123;TARGET_DB&#125; 감지 안 됨...")
        time.sleep(1)
```

  <dl>
    <dt>매핑 함수</dt>
    <dd>한 범위 → 다른 범위 변환. <code>map_value(rssi, -90, -30, 1.0, 0.05)</code></dd>
    <dt>이동 평균</dt>
    <dd>노이즈를 줄이는 기법. 최근 N개의 평균. 부드러움 vs 반응속도 트레이드오프</dd>
    <dt>파라미터 튜닝</dt>
    <dd>정답은 없다! 용도에 맞게 최적화. 한 번에 하나씩 바꿔서 실험 (변인 통제)</dd>
    <dt>LED 출력</dt>
    <dd>데이터를 직관적 피드백으로! 화면 없이도 정보를 전달할 수 있다</dd>
    <dt>MovingAverage 클래스</dt>
    <dd>데이터와 기능을 묶어놓은 구조. <code>add()</code>로 넣고 <code>average()</code>로 평균</dd>
    <dt>clamp (범위 제한)</dt>
    <dd><code>max(min_val, min(max_val, value))</code> — 값을 범위 안으로 제한</dd>
  </dl>

### 흔한 문제 &amp; 해결

<div class="trouble-cards">
  <div class="trouble-item">
    <div class="trouble-symptom">LED가 전혀 깜빡이지 않음</div>
    <div class="trouble-fix">TARGET_DB 이름이 AP측 SSID와 정확히 일치하는지 확인. 대소문자 구분됨!</div>
  </div>
  <div class="trouble-item">
    <div class="trouble-symptom">LED가 항상 느리게만 깜빡임</div>
    <div class="trouble-fix">RSSI_MAX 값 확인. -30이 기본인데 실제로 -30까지 안 나올 수 있음. -40이나 -45로 조정</div>
  </div>
  <div class="trouble-item">
    <div class="trouble-symptom">이동 평균 적용 후 반응이 너무 느림</div>
    <div class="trouble-fix">WINDOW_SIZE를 3으로 줄이기. scan() 자체가 2-3초 걸리므로 window=10이면 20-30초 지연</div>
  </div>
  <div class="trouble-item">
    <div class="trouble-symptom">"감지 안 됨"만 계속 나옴</div>
    <div class="trouble-fix">짝꿍의 AP 피코가 켜져 있는지, 보조배터리가 충전되어 있는지 확인</div>
  </div>
  <div class="trouble-item">
    <div class="trouble-symptom">scan() 에러 발생</div>
    <div class="trouble-fix">STA 모드가 활성화되었는지 확인. <code>sta.active(True)</code> 재실행</div>
  </div>
</div>

```python
# 드래곤볼 AP 모드 -- 내 피코를 드래곤볼로!
# ============================================
# 이 코드는 피코를 Wi-Fi 공유기(AP)로 만들어서
# "DB1" 같은 이름의 Wi-Fi를 생성합니다.
# ============================================

import network
import time

# --- 설정: 내 드래곤볼 이름을 정하세요! ---
DRAGONBALL_NAME = "DB1"   # 여기를 바꿔보세요! (DB1 ~ DB7)
DRAGONBALL_PASSWORD = "dragonball"   # 비밀번호 (8자 이상 필수)

# === 1단계: AP 모드 켜기 ===
# AP = Access Point, 즉 "공유기 모드"
ap = network.WLAN(network.AP_IF)     # AP 인터페이스 가져오기
ap.active(True)                       # AP 활성화

# === 2단계: Wi-Fi 이름과 비밀번호 설정 ===
ap.config(
    essid=DRAGONBALL_NAME,            # Wi-Fi 이름 (SSID)
    password=DRAGONBALL_PASSWORD       # 비밀번호
)

# === 3단계: AP가 잘 켜졌는지 확인 ===
while not ap.active():
    print("AP 켜는 중...")
    time.sleep(1)

# 성공!
print("=" * 40)
print(f"드래곤볼 [&#123;DRAGONBALL_NAME&#125;] 활성화!")
print(f"   IP 주소: &#123;ap.ifconfig()[0]&#125;")
print(f"   비밀번호: &#123;DRAGONBALL_PASSWORD&#125;")
print("=" * 40)
print()
print("신호를 보내고 있습니다...")
print("   (레이더가 찾을 때까지 계속 켜두세요)")
print("   (종료하려면 Ctrl+C)")

# === 4단계: 계속 살아있기 ===
# AP는 코드가 끝나도 유지되지만,
# 상태를 확인할 수 있도록 루프를 돌립니다.
while True:
    # 현재 접속한 기기 수 확인
    stations = ap.status('stations')
    if stations:
        print(f"   -> &#123;len(stations)&#125;대의 기기가 접속 중!")
    time.sleep(5)
```

```python
# ============================================
# 드래곤볼 스캐너 -- 주변 Wi-Fi 탐색기
# ============================================
# STA 모드로 주변의 모든 Wi-Fi를 스캔하고
# 이름, 신호 세기(RSSI), 채널 정보를 보여줍니다.
# ============================================

import network
import time

# === 1단계: STA 모드 켜기 ===
# STA = Station, 즉 "손님 모드"
sta = network.WLAN(network.STA_IF)    # STA 인터페이스 가져오기
sta.active(True)                       # STA 활성화

print("Wi-Fi 스캐너 시작!")
print("=" * 50)

# === 2단계: 반복 스캔 ===
while True:
    print(f"\n스캔 중... (&#123;time.ticks_ms()&#125;ms)")
    print("-" * 50)

    # scan()은 주변 Wi-Fi 목록을 리스트로 돌려줌
    # 각 항목: (ssid, bssid, channel, RSSI, security, hidden)
    networks = sta.scan()

    # === 3단계: 결과 출력 ===
    # RSSI 기준으로 정렬 (신호 강한 순)
    networks.sort(key=lambda x: x[3], reverse=True)

    for i, net in enumerate(networks):
        ssid = net[0].decode('utf-8')   # 바이트 -> 문자열 변환
        rssi = net[3]                    # 신호 세기 (음수, 0에 가까울수록 강함)
        channel = net[2]                 # 채널 번호

        # 신호 세기를 막대로 시각화
        # RSSI: -30(최강) ~ -90(최약) 범위
        bar_length = max(0, (rssi + 90))  # 0~60 범위로 변환
        bar = "=" * (bar_length // 3)     # 막대 그리기

        # 드래곤볼이면 특별 표시!
        if ssid.startswith("DB"):
            marker = "[DB]"
        else:
            marker = "    "

        print(f"  &#123;marker&#125; &#123;ssid:20s&#125; | &#123;rssi:4d&#125;dBm | ch&#123;channel:2d&#125; | &#123;bar&#125;")

    print(f"\n  총 &#123;len(networks)&#125;개 발견")
    print("  (3초 후 다시 스캔...)")
    time.sleep(3)
```

```python
# ============================================
# 드래곤볼 탐지기 -- DB만 찾아내기!
# ============================================
# 주변 Wi-Fi 중 드래곤볼(DB로 시작하는 것)만
# 골라서 보여주는 필터링된 스캐너입니다.
# ============================================

import network
import time

# === STA 모드 켜기 ===
sta = network.WLAN(network.STA_IF)
sta.active(True)

# --- 설정: 드래곤볼 이름 접두사 ---
DB_PREFIX = "DB"    # 이걸로 시작하는 SSID만 찾기

print("드래곤볼 탐지기 가동!")
print("=" * 50)

# === 스캔 루프 ===
while True:
    networks = sta.scan()

    # 핵심: 드래곤볼만 필터링!
    dragonballs = []
    for net in networks:
        ssid = net[0].decode('utf-8')
        rssi = net[3]
        if ssid.startswith(DB_PREFIX):    # "DB"로 시작하면 드래곤볼!
            dragonballs.append((ssid, rssi))

    # RSSI 순으로 정렬 (가까운 순)
    dragonballs.sort(key=lambda x: x[1], reverse=True)

    # 결과 출력
    print(f"\n스캔 결과 -- 드래곤볼 &#123;len(dragonballs)&#125;개 감지!")
    print("-" * 40)

    if dragonballs:
        for ssid, rssi in dragonballs:
            # 거리감 표현
            if rssi > -40:
                distance = "바로 옆!"
            elif rssi > -55:
                distance = "가깝다!"
            elif rssi > -70:
                distance = "중간"
            else:
                distance = "멀다..."

            bar_length = max(0, (rssi + 90)) // 3
            bar = "=" * bar_length

            print(f"  [DB] &#123;ssid:6s&#125; | &#123;rssi:4d&#125;dBm | &#123;bar:20s&#125; | &#123;distance&#125;")
    else:
        print("  드래곤볼을 감지하지 못했습니다...")
        print("  (드래곤볼 AP가 켜져있는지 확인하세요)")

    time.sleep(2)
```

```python
import network
import time

wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect("WiFi이름", "WiFi비밀번호")

print("WiFi 연결 중...", end="")
timeout = 20
while not wlan.isconnected() and timeout > 0:
    print(".", end="")
    time.sleep(1)
    timeout -= 1

if wlan.isconnected():
    print("\nWiFi 연결 완료!", wlan.ifconfig()[0])
else:
    print("\nWiFi 연결 실패!")
```

```javascript
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet()
    ? SpreadsheetApp.getActiveSpreadsheet().getActiveSheet()
    : SpreadsheetApp.openById("시트_ID").getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  var row = [new Date()];
  var keys = Object.keys(data);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["timestamp", ...keys]);
  }
  keys.forEach(function(key) { row.push(data[key]); });
  sheet.appendRow(row);
  return ContentService.createTextOutput(JSON.stringify({status: "ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
```

```python
import urequests

SHEET_URL = "여기에_배포_URL_붙여넣기"

def send_to_sheets(data_dict):
    if not wlan.isconnected():
        return False
    try:
        res = urequests.post(
            SHEET_URL,
            json=data_dict,
            headers={"Content-Type": "application/json"}
        )
        if res.status_code in (301, 302):
            redirect_url = res.headers.get("Location", "")
            res.close()
            if redirect_url:
                res = urequests.post(
                    redirect_url,
                    json=data_dict,
                    headers={"Content-Type": "application/json"}
                )
        print(f"전송 완료: {res.status_code}")
        res.close()
        return True
    except Exception as e:
        print(f"전송 실패: {e}")
        return False

# 테스트 전송
send_to_sheets({"test": "hello", "value": 42})
```

  <p>"DHT20 온습도 센서와 OLED를 I2C에 연결하고, 소리 센서 또는 빛 센서를 ADC 포트에 연결합니다."</p>
  <p>"I2C 포트가 부족하면 I2C Hub를 사용하세요. DHT20(0x38)과 OLED(0x3C)는 주소가 다르므로 Hub 하나로 같은 버스에서 공존 가능합니다."</p>

<div class="student-voice">

<details class="student-content">
  <summary>I2C 스캔 + ADC 테스트 코드</summary>
  <div class="content">

```python
from machine import I2C, Pin, ADC

# I2C 장치 확인
i2c = I2C(1, sda=Pin(6), scl=Pin(7))
devices = i2c.scan()
print("I2C 장치:", [hex(d) for d in devices])
# 예상: ['0x38', '0x3c']

# ADC 센서 테스트
light = ADC(Pin(26))  # 빛 센서 (A0)
sound = ADC(Pin(27))  # 소리 센서 (A1)
print(f"빛: {light.read_u16()}, 소리: {sound.read_u16()}")
```

    <span class="value">Grove 케이블이 제대로 꽂히지 않았거나, I2C 버스가 다른 경우</span>

<details class="student-content">
  <summary>DHT20 + 빛/소리 센서 + OLED 기본 코드</summary>
  <div class="content">

```python
from machine import I2C, Pin, ADC
import time

# I2C 설정 (DHT20 + OLED)
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
dht_addr = 0x38
light = ADC(Pin(26))
sound = ADC(Pin(27))
time.sleep(0.1)

from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

def read_dht20():
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humi = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temp = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temp, 1), round(humi, 1)

while True:
    temp, humi = read_dht20()
    light_pct = light.read_u16() / 65535 * 100
    noise_pct = sound.read_u16() / 65535 * 100

    oled.fill(0)
    oled.fill_rect(0, 0, 128, 12, 1)
    oled.text("Dashboard", 28, 2, 0)
    oled.text(f"T:{temp}C", 4, 16)
    oled.text(f"H:{humi}%", 4, 26)
    oled.text(f"L:{light_pct:.0f}%", 4, 36)
    oled.text(f"N:{noise_pct:.0f}%", 4, 46)
    oled.show()

    print(f"T:{temp} H:{humi} L:{light_pct:.0f} N:{noise_pct:.0f}")
    time.sleep(2)
```

```python
def judge(temp, humi, light_pct, noise_pct):
    if temp > 28 or humi > 70:
        return "환기 필요"
    elif light_pct < 30 and noise_pct < 20:
        return "조용하고 어두움"
    elif noise_pct > 60:
        return "너무 시끄러움"
    else:
        return "쾌적"
```
